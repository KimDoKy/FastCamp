# Python 미흡한 부분 추가 스터디


##### join 

```python
food = ["123","자장면","짬뽕","탕수육"]
   
','.join(food)
'123,자장면,짬뽕,탕수육'

'/'.join(food)
'123/자장면/짬뽕/탕수육'
```

##### split

```python
>>>b = 'a,s,d,f,v,f.4.s.d.d'

>>> b.split('.')
... ['a,s,d,f,v,f', '4', 's', 'd', 'd']
>>> b.split(',')
... ['a', 's', 'd', 'f', 'v', 'f.4.s.d.d']
```
